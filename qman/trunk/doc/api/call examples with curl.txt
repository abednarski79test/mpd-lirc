# get list of queues
curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://localhost:8080/Qman/queues
curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://api.dev.queue.appbucket.eu/queues
	
# get queue details
curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://localhost:8080/Qman/queues/1

curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://api.dev.queue.appbucket.eu/queues/1

# get client status
curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://localhost:8080/Qman/queues/1/tickets/2
	
curl -i -H \
	"Content-Type: application/json" \
	-X GET \
	http://api.dev.queue.appbucket.eu/queues/1/tickets/3
